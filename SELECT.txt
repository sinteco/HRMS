SELECT `e`.`firstname`,`p`.`project_name`, `t`.`task`, CONCAT(FLOOR(SUM( TIME_TO_SEC( IF(sun_date BETWEEN '2020-01-01' AND '2020-07-11',sun_duration,'00:00')) + TIME_TO_SEC( IF(mon_date BETWEEN '2020-01-01' AND '2020-07-11',mon_duration,'00:00')) + TIME_TO_SEC( IF(tue_date BETWEEN '2020-01-01' AND '2020-07-11',tue_duration,'00:00')) + TIME_TO_SEC( IF(wed_date BETWEEN '2020-01-01' AND '2020-07-11',wed_duration,'00:00')) + TIME_TO_SEC( IF(thu_date BETWEEN '2020-01-01' AND '2020-07-11',thu_duration,'00:00')) + TIME_TO_SEC( IF(fri_date BETWEEN '2020-01-01' AND '2020-07-11',fri_duration,'00:00')) + TIME_TO_SEC( IF(sat_date BETWEEN '2020-01-01' AND '2020-07-11',sat_duration,'00:00')))/3600),':', LPAD(FLOOR(SUM( TIME_TO_SEC( IF(sun_date BETWEEN '2020-01-01' AND '2020-07-11',sun_duration,'00:00')) + TIME_TO_SEC( IF(mon_date BETWEEN '2020-01-01' AND '2020-07-11',mon_duration,'00:00')) + TIME_TO_SEC( IF(tue_date BETWEEN '2020-01-01' AND '2020-07-11',tue_duration,'00:00')) + TIME_TO_SEC( IF(wed_date BETWEEN '2020-01-01' AND '2020-07-11',wed_duration,'00:00')) + TIME_TO_SEC( IF(thu_date BETWEEN '2020-01-01' AND '2020-07-11',thu_duration,'00:00')) + TIME_TO_SEC( IF(fri_date BETWEEN '2020-01-01' AND '2020-07-11',fri_duration,'00:00')) + TIME_TO_SEC( IF(sat_date BETWEEN '2020-01-01' AND '2020-07-11',sat_duration,'00:00')))/60)%60,2,'0')) AS `duration` 
FROM `tm_emp_timesheets` AS `et`
INNER JOIN `tm_project_tasks` AS `pt` ON pt.id = et.project_task_id
INNER JOIN `tm_tasks` AS `t` ON t.id = pt.task_id
INNER JOIN `tm_projects` AS `p` ON p.id = pt.project_id and p.id = et.project_id
INNER JOIN `main_employees_summary` AS `e` ON e.user_id = et.emp_id
LEFT JOIN `tm_project_employees` AS `pm` ON p.id = pm.project_id and pm.emp_id = et.emp_id WHERE (et.is_active=1 AND (sun_date BETWEEN '2020-01-01' AND '2020-07-11' OR mon_date BETWEEN '2020-01-01' AND '2020-07-11' OR tue_date BETWEEN '2020-01-01' AND '2020-07-11' OR wed_date BETWEEN '2020-01-01' AND '2020-07-11' OR thu_date BETWEEN '2020-01-01' AND '2020-07-11' OR fri_date BETWEEN '2020-01-01' AND '2020-07-11' OR sat_date BETWEEN '2020-01-01' AND '2020-07-11')) GROUP BY `t`.`id`,
`p`.`id`, `e`.`id`
ORDER by `e`.`firstname`
